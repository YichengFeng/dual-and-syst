185 double get_uncertainty() const {
186 double sum2 = 0.0;
187 for (const auto& p : duals_) sum2 += p.second * p.second;
188 if (correlations_.empty()) return std::sqrt(sum2);
189 for (const auto& c : correlations_) {
190 int i = c.first.first, j = c.first.second;
191 double rho = c.second;
192 auto it_i = duals_.find(i);
193 auto it_j = duals_.find(j);
194 if (it_i != duals_.end() && it_j != duals_.end()) {
195 sum2 += 2.0 * it_i->second * it_j->second * rho;
196 }
197 }
198 return std::sqrt(sum2);
199 }

245 while (ia != a.duals_.end() && ib != b.duals_.end()) {
246 if (ia->first < ib->first) {

250 result.duals_[ib->first] = ib->second;

252 } else {
253 result.duals_[ia->first] = ia->second + ib->second;
254 ++ia; ++ib;
255 }

258 for (; ib != b.duals_.end(); ++ib) result.duals_[ib->first] = ib->second;

428 friend DualMultiv atan2(const DualMultiv& y, const DualMultiv& x) {
429 double denom = x.value_ * x.value_ + y.value_ * y.value_;
430 DualMultiv result(std::atan2(y.value_, x.value_));
431 auto iy = y.duals_.begin(), ix = x.duals_.begin();
432 while (iy != y.duals_.end() && ix != x.duals_.end()) {
433 if (iy->first < ix->first) {
434 result.duals_[iy->first] = (x.value_ * iy->second) / denom;
435 ++iy;
436 } else if (iy->first > ix->first) {
437 result.duals_[ix->first] = (-y.value_ * ix->second) / denom;
438 ++ix;
439 } else {
440 result.duals_[iy->first] = (x.value_ * iy->second - y.value_ * ix->second) / denom;
441 ++iy; ++ix;
442 }
443 }
444 for (; iy != y.duals_.end(); ++iy)
445 result.duals_[iy->first] = (x.value_ * iy->second) / denom;
446 for (; ix != x.duals_.end(); ++ix)
447 result.duals_[ix->first] = (-y.value_ * ix->second) / denom;
