55enum class SystMergeMode {
56 // ---- one-sided ----
57 kOneSided_Simple,
58 kOneSided_Barlow_Subset,
59 kOneSided_Barlow_Independent,
60 kOneSided_Barlow_FullyCorrelated,
61
62 // ---- two-sided ----
63 kTwoSided_Simple,
64 kTwoSided_Barlow_Subset,
65 kTwoSided_Barlow_Independent,
66 kTwoSided_Barlow_FullyCorrelated,
67
68 kNull
69};

73inline bool is_one_sided(SystMergeMode sm) noexcept {
74 return (sm == SystMergeMode::kOneSided_Simple ||
75 sm == SystMergeMode::kOneSided_Barlow_Subset ||
76 sm == SystMergeMode::kOneSided_Barlow_Independent ||
77 sm == SystMergeMode::kOneSided_Barlow_FullyCorrelated);
78}

365inline SystResult SystGraph::evaluate_systematics(const std::vector<int>& vids) const
366{
367 SystResult result;
368
369 std::vector<int> eval_vids;
370 for (int vid : vids) {
371 if (var_.count(vid)) {
372 eval_vids.push_back(vid);
373 } else {
374 std::cerr << "SystGraph::evaluate_systematics(): vid "
375 << vid << " not found, skipped." << std::endl;
376 }
377 }
378
379 const int nvar = static_cast<int>(eval_vids.size());
380 const int nbin = def_.get_n();
381 if (nvar == 0 || nbin == 0) return result;
382
383 result.vids = eval_vids;
384 result.var_syst.assign(nvar, std::vector<double>(nbin, 0.0));
385 result.var_stat.assign(nvar, std::vector<double>(nbin, 0.0));
386 result.var_diff.assign(nvar, std::vector<double>(nbin, 0.0));
387 result.total_lower.assign(nbin, 0.0);
388 result.total_upper.assign(nbin, 0.0);
389
390 for (int iv = 0; iv < nvar; ++iv) {
391 int vid = eval_vids[iv];
392 const SystVariation& sv = var_.find(vid)->second;
393 const DualGraph& vg = sv.graph;
394 SystMergeMode sm = (merge_mode_ != SystMergeMode::kNull) ? merge_mode_ : sv.mode;
395 double w = sv.weight;
396
397 if (sm == SystMergeMode::kNull) continue;
398
399 const bool one_sided = ::is_one_sided(sm);
400
401 for (int ib = 0; ib < nbin; ++ib) {
402 double d = def_.get_point(ib).get_py().get_value();
403 double ed = def_.get_point(ib).get_py().get_uncertainty();
404 double v = vg.get_point(ib).get_py().get_value();
405 double ev = vg.get_point(ib).get_py().get_uncertainty();
406
407 double diff = v - d;
408 result.var_diff[iv][ib] = diff;
409 double raw = diff;
410 double stat = 0.0;
411
412 switch (sm) {
413 case SystMergeMode::kOneSided_Barlow_Subset:
414 case SystMergeMode::kTwoSided_Barlow_Subset: {
415 double dvar = std::fabs(ev * ev - ed * ed);
416 stat = std::sqrt(dvar);
417 double arg = diff * diff - dvar;
418 raw = (arg > 0.0) ? std::copysign(std::sqrt(arg), diff) : 0.0;
419 break;
420 }
421 case SystMergeMode::kOneSided_Barlow_Independent:
422 case SystMergeMode::kTwoSided_Barlow_Independent: {
423 double dvar = ev * ev + ed * ed;
424 stat = std::sqrt(dvar);
425 double arg = diff * diff - dvar;
426 raw = (arg > 0.0) ? std::copysign(std::sqrt(arg), diff) : 0.0;
427 break;
428 }
429 case SystMergeMode::kOneSided_Barlow_FullyCorrelated:
430 case SystMergeMode::kTwoSided_Barlow_FullyCorrelated: {
431 stat = std::fabs(ev - ed);
432 double arg = diff * diff - stat * stat;
433 raw = (arg > 0.0) ? std::copysign(std::sqrt(arg), diff) : 0.0;
434 break;
435 }
436 default:
437 break;
438 }
439
440 double syst = (one_sided) ? raw : std::fabs(raw);
441 result.var_syst[iv][ib] = syst;
442 result.var_stat[iv][ib] = stat;
443
444 if (one_sided) {
445 if (syst > 0.0) result.total_upper[ib] += w * syst * syst;
446 else result.total_lower[ib] += w * syst * syst;
447 } else {
448 result.total_lower[ib] += w * syst * syst;
449 result.total_upper[ib] += w * syst * syst;
450 }
451 }
452 }
453
454 for (int ib = 0; ib < nbin; ++ib) {
455 result.total_lower[ib] = std::sqrt(result.total_lower[ib]);
456 result.total_upper[ib] = std::sqrt(result.total_upper[ib]);
457 }
458
459 return result;
460}

466inline std::vector<std::string> SystGraph::to_latex(bool transpose) const
467{
468 SystResult result = evaluate_systematics();
469 const int nvar = static_cast<int>(result.vids.size());
470 const int nbin = static_cast<int>(result.total_lower.size());
471
472 std::vector<std::string> lines;
473 if (nvar == 0 || nbin == 0) return lines;
474
475 auto fmt_num = [](double v) -> std::string {
476 if (v == 0.0) return "0";
477 ValueErrors ve(v);
478 return ve.print_err0("L");
479 };
480
481 auto fmt_val = [](double v) -> std::string {
482 if (v == 0.0) return "0";
483 ValueErrors ve(v);
484 if (ve.get_precision() < 3) ve.set_precision(3);
485 return ve.print_err0("L");
486 };
487
488 auto fmt_diff = [&](double diff) -> std::string {
489 if (diff == 0.0) return "0";
490 double ad = std::fabs(diff);
491 if (diff > 0.0)
492 return "$+" + fmt_num(ad) + "$";
493 else
494 return "$-" + fmt_num(ad) + "$";
495 };
496
497 auto fmt_cell = [&](double lo, double hi) -> std::string {
498 if (lo == 0.0 && hi == 0.0) return "0";
499 double max_err = std::max(std::fabs(lo), std::fabs(hi));
500 ValueErrors ve_ref(0.0, max_err);
501 int prec = ve_ref.get_precision();
502
503 auto fmt_side = [prec](double v) -> std::string {
504 ValueErrors ve(v);
505 ve.set_precision(prec);
506 return ve.print_err0("L");
507 };
508
509 if (lo == hi) {
510 return "$\\pm " + fmt_side(lo) + "$";
511 } else if (lo == 0.0) {
512 return "$+" + fmt_side(hi) + "$";
513 } else if (hi == 0.0) {
514 return "$-" + fmt_side(lo) + "$";
515 } else {
516 return "$_{-" + fmt_side(lo)
517 + "}^{+" + fmt_side(hi) + "}$";
518 }
519 };
520
521 std::vector<std::string> names(nvar);
522 std::vector<std::string> weights(nvar);
523 std::vector<std::string> sides(nvar);
524 for (int iv = 0; iv < nvar; ++iv) {
525 auto it = var_.find(result.vids[iv]);
526 if (it == var_.end()) {
527 names[iv] = "vid_" + std::to_string(result.vids[iv]);
528 weights[iv] = "1";
529 sides[iv] = "?";
530 } else {
531 SystMergeMode sm = (merge_mode_ != SystMergeMode::kNull)
532 ? merge_mode_ : it->second.mode;
533 names[iv] = it->second.name;
534 {
535 std::ostringstream ws;
536 ws << it->second.weight;
537 weights[iv] = ws.str();
538 }
539 bool one_sided = ::is_one_sided(sm);
540 sides[iv] = one_sided ? "1-sided" : "2-sided";
541 }
542 }
543
544 auto bin_label = [&](int ib) -> std::string {
545 double x = def_.get_point(ib).get_px().get_value();
546 ValueErrors ve(x);
547 return ve.print_err0("L", true);
548 };
549
550 if (!transpose) {
551 std::string coldef = "l|cc|";
552 for (int ib = 0; ib < nbin; ++ib) coldef += "c";
553 lines.push_back("\\begin{tabular}{" + coldef + "}");
554 lines.push_back("\\hline");
555
556 std::string hdr = "Variation & Weight & Side";
557 for (int ib = 0; ib < nbin; ++ib)
558 hdr += " & " + bin_label(ib);
559 hdr += " \\\\";
560 lines.push_back(hdr);
561 lines.push_back("\\hline");
562
563 std::string def_row = "Default & & & ";
564 for (int ib = 0; ib < nbin; ++ib) {
565 if (ib > 0) def_row += " & ";
566 def_row += fmt_val(def_.get_point(ib).get_py().get_value());
567 }
568 def_row += " \\\\";
569 lines.push_back(def_row);
570
571 std::string stat_row = "Stat. unc. & & & ";
572 for (int ib = 0; ib < nbin; ++ib) {
573 if (ib > 0) stat_row += " & ";
574 stat_row += fmt_num(def_.get_point(ib).get_py().get_uncertainty());
575 }
576 stat_row += " \\\\";
577 lines.push_back(stat_row);
578
579 lines.push_back("\\hline");
580 for (int iv = 0; iv < nvar; ++iv) {
581 std::string row = names[iv] + " & " + weights[iv]
582 + " & " + sides[iv];
583 for (int ib = 0; ib < nbin; ++ib)
584 row += " & " + fmt_diff(result.var_diff[iv][ib]);
585 row += " \\\\";
586 lines.push_back(row);
587 }
588
589 lines.push_back("\\hline");
590 std::string tot = "Total & & & ";
591 for (int ib = 0; ib < nbin; ++ib) {
592 if (ib > 0) tot += " & ";
593 tot += fmt_cell(result.total_lower[ib],
594 result.total_upper[ib]);
595 }
596 tot += " \\\\";
597 lines.push_back(tot);
598 lines.push_back("\\hline");
599
600 lines.push_back("\\end{tabular}");
601
602 } else {
603 std::string coldef = "l|cc|";
604 for (int iv = 0; iv < nvar; ++iv) coldef += "c";
605 coldef += "c";
606 lines.push_back("\\begin{tabular}{" + coldef + "}");
607 lines.push_back("\\hline");
608
609 std::string hdr1 = "Bin & Default & Stat. unc.";
610 for (int iv = 0; iv < nvar; ++iv)
611 hdr1 += " & " + names[iv];
612 hdr1 += " & Total \\\\";
613 lines.push_back(hdr1);
614 lines.push_back("\\hline");
615
616 std::string hdr2 = " & & ";
617 for (int iv = 0; iv < nvar; ++iv)
618 hdr2 += " & " + weights[iv];
619 hdr2 += " & \\\\";
620 lines.push_back(hdr2);
621
622 std::string hdr3 = " & & ";
623 for (int iv = 0; iv < nvar; ++iv)
624 hdr3 += " & " + sides[iv];
625 hdr3 += " & \\\\";
626 lines.push_back(hdr3);
627
628 lines.push_back("\\hline");
629
630 for (int ib = 0; ib < nbin; ++ib) {
631 std::string row = bin_label(ib);
632 row += " & " + fmt_val(def_.get_point(ib).get_py().get_value());
633 row += " & " + fmt_num(def_.get_point(ib).get_py().get_uncertainty());
634 for (int iv = 0; iv < nvar; ++iv)
635 row += " & " + fmt_diff(result.var_diff[iv][ib]);
636 row += " & " + fmt_cell(result.total_lower[ib],
637 result.total_upper[ib]);
638 row += " \\\\";
639 lines.push_back(row);
640 }
641
642 lines.push_back("\\hline");
643 lines.push_back("\\end{tabular}");
644 }
645
646 return lines;
647}

658template <typename Op>
659inline SystGraph apply_binary(const SystGraph& a, const SystGraph& b, Op op) {
660 SystGraph out(op(a.get_def(), b.get_def()));
661 const auto& va = a.get_var_all();
662 const auto& vb = b.get_var_all();
663 auto ia = va.begin();
664 auto ib = vb.begin();
665 while (ia != va.end() && ib != vb.end()) {
666 if (ia->first < ib->first) {
667 SystVariation sv = ia->second;
668 sv.graph = op(ia->second.graph, b.get_def());
669 out.add_var(ia->first, sv);
670 ++ia;
671 } else if (ia->first > ib->first) {
672 SystVariation sv = ib->second;
673 sv.graph = op(a.get_def(), ib->second.graph);
674 out.add_var(ib->first, sv);
675 ++ib;
676 } else {
677 SystVariation sv = ia->second;
678 sv.graph = op(ia->second.graph, ib->second.graph);
679 out.add_var(ia->first, sv);
680 ++ia; ++ib;
681 }
682 }
683 for (; ia != va.end(); ++ia) {
684 SystVariation sv = ia->second;
685 sv.graph = op(ia->second.graph, b.get_def());
686 out.add_var(ia->first, sv);
687 }
688 for (; ib != vb.end(); ++ib) {
689 SystVariation sv = ib->second;
690 sv.graph = op(a.get_def(), ib->second.graph);
691 out.add_var(ib->first, sv);
692 }
693 return out;
694}

697template <typename Op>
698inline SystGraph apply_unary(const SystGraph& a, Op op) {
699 SystGraph out(op(a.get_def()));
700 for (const auto& kv : a.get_var_all()) {
701 SystVariation sv = kv.second;
702 sv.graph = op(kv.second.graph);
703 out.add_var(kv.first, sv);
704 }
705 return out;
706}
