234 void calc() {
235 int n = static_cast<int>(points_.size());
236 std::vector<double> vx(n), ex(n), vy(n), ey(n);
237 for (int i = 0; i < n; ++i) {
238 vx[i] = points_[i].get_px().get_value();
239 ex[i] = points_[i].get_px().get_derivative();
240 vy[i] = points_[i].get_py().get_value();
241 ey[i] = points_[i].get_py().get_uncertainty();
242 }
243 graph_ = TGraphErrors(n, vx.data(), vy.data(), ex.data(), ey.data());
244 is_updated_ = true;
245 }

265 // --- hide bad points ---
266
270 void hide_bad_points() {
271 const int n = graph_.GetN();
272 double *x = graph_.GetX();
273 double *y = graph_.GetY();
274 double *ex = graph_.GetEX();
275 double *ey = graph_.GetEY();
276
277 std::vector<double> vx, vex, vy, vey;
278 vx.reserve(n);
279 vex.reserve(n);
280 vy.reserve(n);
281 vey.reserve(n);
282
283 for (int i = 0; i < n; i++) {
284 if (y[i] == 0 && ey[i] == 0) continue;
285 if (std::isnan(x[i]) || std::isinf(x[i]) ||
286 std::isnan(y[i]) || std::isinf(y[i]) ||
287 std::isnan(ex[i]) || std::isinf(ex[i]) ||
288 std::isnan(ey[i]) || std::isinf(ey[i])) continue;
289 vx.push_back(x[i]);
290 vex.push_back(ex[i]);
291 vy.push_back(y[i]);
292 vey.push_back(ey[i]);
293 }
294
295 const int n2 = static_cast<int>(vx.size());
296 if (n2 == n) return; // nothing removed
297
298 graph_.Set(n2);
299 for (int i = 0; i < n2; i++) {
300 graph_.GetX()[i] = vx[i];
301 graph_.GetEX()[i] = vex[i];
302 graph_.GetY()[i] = vy[i];
303 graph_.GetEY()[i] = vey[i];
304 }
305 }
