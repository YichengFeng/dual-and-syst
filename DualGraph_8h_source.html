199 DualGraph average_bin(const std::vector<int>& range) const {
200 std::vector<int> unique_range = range;
201 std::sort(unique_range.begin(), unique_range.end());
202 unique_range.erase(
203 std::unique(unique_range.begin(), unique_range.end()),
204 unique_range.end());
205 int n = static_cast<int>(unique_range.size());
206 if (n <= 0) {
207 std::cerr << "DualGraph::average_bin(): invalid range!" << std::endl;
208 return DualGraph();
209 }
210 if (!check_index(unique_range[0]) || !check_index(unique_range[n - 1])) {
211 std::cerr << "DualGraph::average_bin(): invalid range!" << std::endl;
212 return DualGraph();
213 }
214 DualNumber avex(0.0, 0.0);
215 DualMultiv avey;
216 for (int k = 0; k < n; ++k) {
217 int i = unique_range[k];
218 avex = avex + points_[i].get_px();
219 DualMultiv dm;
220 dm.set_value(points_[i].get_py().get_value());
221 std::vector<int> ids = points_[i].get_py().get_ids();
222 for (std::size_t j = 0; j < ids.size(); ++j) {
223 int idx = ids[j];
224 dm.set_derivative(idx, points_[i].get_py().get_derivative(idx));
225 }
226 if (k == 0) {
227 avey = dm;
228 } else {
229 double e1 = avey.get_uncertainty();
230 double e2 = dm.get_uncertainty();
231 double w1 = 1.0 / (e1 * e1);
232 double w2 = 1.0 / (e2 * e2);
233 avey = (w1 * avey + w2 * dm) / (w1 + w2);
234 }
235 }
236 avex = avex / static_cast<double>(n);
237 std::vector<DualPoint> out;
238 out.emplace_back(avex, avey);
239 return DualGraph(out);
240 }

260 DualNumber avex(0.0, 0.0);

262 double inv_n = 1.0 / n;
263 for (int k = 0; k < n; ++k) {
264 int i = unique_range[k];
265 avex = avex + points_[i].get_px();
266 DualMultiv dm;
267 dm.set_value(points_[i].get_py().get_value());
268 std::vector<int> ids = points_[i].get_py().get_ids();
269 for (std::size_t j = 0; j < ids.size(); ++j) {
270 int idx = ids[j];
271 dm.set_derivative(idx, points_[i].get_py().get_derivative(idx));
272 }
273 avey = avey + dm * inv_n;
274 }
275 avex = avex * inv_n;
276 std::vector<DualPoint> out;
277 out.emplace_back(avex, avey);
