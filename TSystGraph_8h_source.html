284 // =====================================================================
285 // calc — evaluate systematics and sync ROOT output graphs
286 // =====================================================================
287
293 void calc(const std::vector<int>& vidlist) {
294 int n = def_.get_n();
295 if (n <= 0) return;
296
297 calc_result_ = evaluate_systematics(vidlist);
298
299 std::vector<double> vx, vex, vy, vey, syl, syh;
300 vx.reserve(n); vex.reserve(n); vy.reserve(n);
301 vey.reserve(n); syl.reserve(n); syh.reserve(n);
302
303 for (int i = 0; i < n; ++i) {
304 double xv = def_.get_point(i).get_px().get_value();
305 if (xv < xlim_low_ || xv >= xlim_high_) continue;
306 vx.push_back(xv);
307 vex.push_back(width_);
308 vy.push_back(def_.get_point(i).get_py().get_value());
309 vey.push_back(def_.get_point(i).get_py().get_uncertainty());
310 syl.push_back(calc_result_.total_lower[i]);
311 syh.push_back(calc_result_.total_upper[i]);
312 }
313
314 int np = static_cast<int>(vx.size());
315
316 std::vector<double> sylc(np), syhc(np);
317 for (int i = 0; i < np; ++i) {
318 sylc[i] = std::sqrt(vey[i] * vey[i] + syl[i] * syl[i]);
319 syhc[i] = std::sqrt(vey[i] * vey[i] + syh[i] * syh[i]);
320 }
321
322 if (stat_.GetN() != np) stat_.Set(np);
323 for (int i = 0; i < np; ++i) {
324 stat_.GetX()[i] = vx[i];
325 stat_.GetY()[i] = vy[i];
326 stat_.GetEX()[i] = vex[i];
327 stat_.GetEY()[i] = vey[i];
328 }
329 stat_.SetHistogram(nullptr);
330
331 if (syst_.GetN() != np) syst_.Set(np);
332 for (int i = 0; i < np; ++i) {
333 syst_.GetX()[i] = vx[i];
334 syst_.GetY()[i] = vy[i];
335 syst_.GetEXlow()[i] = vex[i];
336 syst_.GetEXhigh()[i] = vex[i];
337 syst_.GetEYlow()[i] = syl[i];
338 syst_.GetEYhigh()[i] = syh[i];
339 }
340 syst_.SetHistogram(nullptr);
341
342 if (comb_.GetN() != np) comb_.Set(np);
343 for (int i = 0; i < np; ++i) {
344 comb_.GetX()[i] = vx[i];
345 comb_.GetY()[i] = vy[i];
346 comb_.GetEXlow()[i] = vex[i];
347 comb_.GetEXhigh()[i] = vex[i];
348 comb_.GetEYlow()[i] = sylc[i];
349 comb_.GetEYhigh()[i] = syhc[i];
350 }
351 comb_.SetHistogram(nullptr);
352
353 is_updated_ = true;
354 }

633 // =====================================================================
634 // MakeHist — stacked histogram of systematic^2 breakdown
635 // =====================================================================
636
648 void make_hist(const char* name, TH2D* hFrame,
649 bool is_percent = false,
650 const char* path = "systplot",
651 std::vector<TString> file_types = {"pdf"}) {
652 shift_x();
653
654 TString nm(name);
655 if (is_percent) nm = nm + "Percent";
656
657 int vcol[10] = {2, 3, 4, kOrange, 6, 7, 8, 9, 16, 28};
658 int vfls[10] = {1001, 3001, 3002, 3004, 3244,
659 3003, 3344, 3345, 3444, 3354};
660
661 int nvar = static_cast<int>(var_.size());
662 int ncln = nvar / 20 + 1;
663 int wpxl = 700 + ncln * 175;
664 int hpxl = 500;
665
666 TCanvas* c = new TCanvas(
667 TString("SystHist") + nm,
668 TString("SystHist") + nm, wpxl, hpxl);
669 gPad->SetLeftMargin(0.18 * 700 / wpxl);
670 gPad->SetRightMargin((0.04 * 700 + ncln * 175) / wpxl);
671 gPad->SetTopMargin(0.10);
672 gPad->SetBottomMargin(0.16);
673
674 double xl = hFrame->GetXaxis()->GetXmin();
675 double xh = hFrame->GetXaxis()->GetXmax();
676 TString xtitle = hFrame->GetXaxis()->GetTitle();
677 TString ytitle = TString(hFrame->GetYaxis()->GetTitle())
678 + ": syst^{2}";
679 if (is_percent) ytitle = ytitle + " (%)";
680
681 TGraphErrors gTmp = stat_;
682 gTmp.SetLineColor(kBlack);
683 gTmp.SetMarkerColor(kBlack);
684 gTmp.SetMarkerStyle(20);
685
686 int nBin = gTmp.GetN();
687 if (nBin <= 0) { delete c; return; }
688
689 double bw = std::fabs(xh - xl) / (nBin + 1.0);
690 for (int i = 1; i < nBin; ++i) {
691 double tmpbw = std::fabs(
692 gTmp.GetX()[i] - gTmp.GetX()[i - 1]);
693 if (i == 1) bw = tmpbw;
694 else bw = std::min(bw, tmpbw);
695 }
696
697 bool is_good_xedge = true;
698 std::vector<double> xcenter(nBin);
699 for (int i = 0; i < nBin; ++i)
700 xcenter[i] = gTmp.GetX()[i];
701 std::sort(xcenter.begin(), xcenter.end());
702
703 std::vector<double> xedge(nBin + 1);
704 xedge[0] = xl;
705 for (int i = 0; i < nBin; ++i) {
706 xedge[i + 1] = xedge[i]
707 + (xcenter[i] - xedge[i]) * 2.0;
708 if (xedge[i + 1] <= xedge[i]) is_good_xedge = false;
709 }
710 if (!is_good_xedge) {
711 xedge[0] = xcenter[0] - 0.5 * bw;
712 for (int i = 0; i < nBin; ++i)
713 xedge[i + 1] = xcenter[i] + 0.5 * bw;
714 }
715
716 std::vector<int> xbinedge(nBin);
717 for (int i = 0; i < nBin; ++i) {
718 double xc = gTmp.GetX()[i];
719 for (int j = 0; j < nBin; ++j) {
720 if (xedge[j] <= xc && xedge[j + 1] >= xc) {
721 xbinedge[i] = j;
722 break;
723 }
724 }
725 }
726
727 TLegend* leg = new TLegend(
728 700.0 / wpxl, 0.01,
729 1 - 0.01 * hpxl / wpxl, 0.99);
730 leg->SetFillStyle(0);
731 leg->SetBorderSize(1);
732 leg->SetNColumns(ncln);
733 leg->SetTextSize(0.04);
734
735 int nsy = static_cast<int>(calc_result_.vids.size());
736 if (nsy == 0) { delete c; delete leg; return; }
737
738 std::vector<std::vector<TBox>> syBoxUpper(nsy);
739 std::vector<std::vector<TBox>> syBoxLower(nsy);
740 double syMaxUpper = 0.0;
741 double syMaxLower = 0.0;
742
743 for (int j = 0; j < nBin; ++j) {
744 double s2_total_upper = calc_result_.total_upper[j]
745 * calc_result_.total_upper[j];
746 double s2_total_lower = calc_result_.total_lower[j]
747 * calc_result_.total_lower[j];
748
749 double syBinUpper = 0.0;
750 double syBinLower = 0.0;
751
752 for (int i = nsy - 1; i >= 0; --i) {
753 int vid = calc_result_.vids[i];
754 const SystVariation& sv = var_.find(vid)->second;
755 double w = sv.weight;
756 double syst = calc_result_.var_syst[i][j];
757
758 SystMergeMode sm = (merge_mode_ != SystMergeMode::kNull)
759 ? merge_mode_ : sv.mode;
760 bool one_sided = ::is_one_sided(sm);
761
762 double contrib = w * syst * syst;
763
764 bool to_upper = !one_sided || syst > 0.0;
765 bool to_lower = !one_sided || syst < 0.0;
766
767 int col = vcol[i % 10];
768 int ifls = i / 10;
769 while (ifls >= 10) ifls /= 10;
770 int fls = vfls[ifls];
771
772 if (to_upper && contrib > 0.0) {
773 double upval = contrib;
774 if (is_percent && s2_total_upper != 0.0)
775 upval = 100.0 * upval / s2_total_upper;
776
777 TBox box(xedge[xbinedge[j]], syBinUpper,
778 xedge[xbinedge[j] + 1],
779 syBinUpper + upval);
780 box.SetLineColor(col);
781 box.SetFillStyle(fls);
782 box.SetFillColor(col);
783 syBoxUpper[i].push_back(box);
784 syBinUpper += upval;
785 }
786
787 if (to_lower && contrib > 0.0) {
788 double lowval = contrib;
789 if (is_percent && s2_total_lower != 0.0)
790 lowval = 100.0 * lowval / s2_total_lower;

798 box.SetFillColor(col);

801 }
802 }

804 if (syBinLower > syMaxLower) syMaxLower = syBinLower;
805 }

815 hFrameTmp->GetXaxis()->SetTitle(xtitle);
816 hFrameTmp->GetYaxis()->SetTitle(ytitle);
817 hFrameTmp->GetYaxis()->SetTitleOffset(1.0 / wpxl * 700);
818 hFrameTmp->Draw();
819
820 for (int i = 0; i < nsy; ++i) {
821 int vid = calc_result_.vids[i];
822 const SystVariation& sv = var_.find(vid)->second;
823 if (!syBoxUpper[i].empty())
824 leg->AddEntry(&syBoxUpper[i][0], sv.name.c_str(), "f");
825 else if (!syBoxLower[i].empty())
826 leg->AddEntry(&syBoxLower[i][0], sv.name.c_str(), "f");
827 }
828
829 for (int i = 0; i < nsy; ++i) {
830 for (auto& box : syBoxUpper[i]) box.Draw();
831 for (auto& box : syBoxLower[i]) box.Draw();
832 }
833
834 TLine *tmp0 = new TLine(xl,0, xh,0);
835 tmp0->Draw("same");
836
837 leg->Draw("same");
838 gPad->SetTicks();
839
840 for (const auto& ft : file_types)
841 c->SaveAs(TString(path) + "/Syst" + nm
842 + "Hist." + ft);
843
844 delete hFrameTmp;
845 delete leg;
846 delete c;
