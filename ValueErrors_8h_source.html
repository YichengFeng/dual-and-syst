215 static double calc_coefficient(double x) {
216 if (x == 0.0) return 0.0;
217 double ax = std::fabs(x);
218 while (ax < 1.0) ax *= 10.0;
219 while (ax >= 10.0) ax *= 0.1;
220 return (x > 0.0) ? ax : -ax;
221 }
222
224 static int calc_exponent(double x) {
225 if (x == 0.0) return 0;
226 int n = 0;
227 double ax = std::fabs(x);
228 while (ax < 1.0) { ax *= 10.0; --n; }
229 while (ax >= 10.0) { ax *= 0.1; ++n; }
230 return n;
231 }

236 static FormatResult valu_err1(double val, double err) {
237 FormatResult r = {};
238 if (val == 0.0 && err == 0.0) return r;
239
240 double val_c = calc_coefficient(val);
241 int val_e = calc_exponent(val);
242 double err_c = calc_coefficient(err);
243 int err_e = calc_exponent(err);
244
245 if (val == 0.0) val_e = err_e;
246 if (err == 0.0) err_e = val_e;
247
248 double value, error;
249 int exponent, precision;
250
251 if (std::fabs(val) >= std::fabs(err) || val_e >= err_e) {
252 value = val_c;
253 exponent = val_e;
254 int err_de = err_e - val_e;
255 error = err_c * std::pow(10.0, err_de);
256 precision = (std::fabs(err_c) < 3.5) ? (1 - err_de) : (0 - err_de);
257 } else if (val_e + 1 == err_e && std::fabs(err_c) < 3.5) {
258 value = val_c / 10.0;
259 exponent = err_e;
260 error = err_c;
261 precision = 1;
262 } else if (std::fabs(err_c) < 3.5) {
263 value = val_c * std::pow(10.0, val_e - err_e);
264 exponent = err_e;
265 error = err_c;
266 precision = 1;
267 } else {
268 value = val_c * std::pow(10.0, val_e - err_e);
269 exponent = err_e;
270 error = err_c;
271 precision = 0;
272 }
273
274 r.precision = precision;
275 r.exponent = exponent;
276 r.value_coeff = value;
277 r.stat_coeff = error;
278 return r;
279 }

289 static FormatResult valu_err2(double val, double stat, double syst) {
290 double err = std::fabs(stat) > std::fabs(syst) ? std::fabs(stat) : std::fabs(syst);
291
292 double val_c = calc_coefficient(val);
293 int val_e = calc_exponent(val);
294 double stat_c = calc_coefficient(stat);
295 int stat_e = calc_exponent(stat);
296 double syst_c = calc_coefficient(syst);
297 int syst_e = calc_exponent(syst);
298
299 FormatResult r1 = valu_err1(val, err);
300 int exponent = r1.exponent;
301
302 FormatResult r = {};
303 r.precision = r1.precision;
304 r.exponent = exponent;
305 r.value_coeff = val_c * std::pow(10.0, val_e - exponent);
306 r.stat_coeff = stat_c * std::pow(10.0, stat_e - exponent);
307 r.syst_coeff = syst_c * std::pow(10.0, syst_e - exponent);
308 r.sysl_coeff = r.syst_coeff;
309 return r;
310 }

313 static FormatResult valu_err3(double val, double stat, double sysl, double sysh) {
314 double err = std::fabs(stat);
315 if (std::fabs(sysl) > err) err = std::fabs(sysl);
316 if (std::fabs(sysh) > err) err = std::fabs(sysh);
317
318 double val_c = calc_coefficient(val);
319 int val_e = calc_exponent(val);
320 double stat_c = calc_coefficient(stat);
321 int stat_e = calc_exponent(stat);
322 double sysl_c = calc_coefficient(sysl);
323 int sysl_e = calc_exponent(sysl);
324 double sysh_c = calc_coefficient(sysh);
325 int sysh_e = calc_exponent(sysh);
326
327 FormatResult r1 = valu_err1(val, err);
328 int exponent = r1.exponent;
329
330 double sysl_val = sysl_c * std::pow(10.0, sysl_e - exponent);
331 double sysh_val = sysh_c * std::pow(10.0, sysh_e - exponent);
332
333 FormatResult r = {};
334 r.precision = r1.precision;
335 r.exponent = exponent;
336 r.value_coeff = val_c * std::pow(10.0, val_e - exponent);
337 r.stat_coeff = stat_c * std::pow(10.0, stat_e - exponent);
338 r.syst_coeff = std::sqrt(sysl_val * sysl_val + sysh_val * sysh_val);
339 r.sysl_coeff = sysl_val;
340 r.sysh_coeff = sysh_val;
341 return r;
342 }

345 void recalc(Mode mode) {
346 mode_ = mode;
347 FormatResult r;
348 switch (mode_) {
349 case VALUE_ONLY: r = valu_err0(value_); break;
350 case VALUE_STAT: r = valu_err1(value_, stat_); break;
351 case VALUE_STAT_SYST: r = valu_err2(value_, stat_, syst_); break;
352 case VALUE_STAT_ASYM: r = valu_err3(value_, stat_, sysl_, sysh_); break;
353 default: return;
354 }
355 precision_ = r.precision;
356 exponent_ = r.exponent;
357 value_coeff_ = r.value_coeff;
358 stat_coeff_ = r.stat_coeff;
359 syst_coeff_ = r.syst_coeff;
360 sysl_coeff_ = r.sysl_coeff;
361 sysh_coeff_ = r.sysh_coeff;
362 }
